    public int[] solution(long n) {
        String s = String.valueOf(n); //1 2 3 4 5
        int[] answer = new int[s.length()];
        for(int i=0;i<answer.length;i++)
        {

class Solution {
    boolean solution(String s) {
        boolean answer = true;
        int p=0,y=0;
        s=s.toLowerCase();
        for(int i=0;i<s.length();i++)
        {
            if(s.charAt(i)=='p')p++;
            else if(s.charAt(i)=='y')y++;
        }
        if(p!=y)
            answer = false;

        return answer;
    }
}
