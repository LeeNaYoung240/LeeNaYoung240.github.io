```java
import java.util.*;  
import java.io.*;  
public class Main {  
    public static void main(String[] args) throws IOException {  
       BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));  
          //선언

        }

    }  
}
```

```java
import java.util.*;  
import java.io.*;  
public class Main {  
    public static void main(String[] args) throws IOException {  
        BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));  
  
        StringTokenizer st = new StringTokenizer(bf.readLine()," ");  
        //입력된 한 줄을 공백을 기준으로 나누어 저장  
  int N = Integer.parseInt(st.nextToken());  
        //첫 번째 토큰을 정수형으로 변환하여 변수 N에 저장  
  int X = Integer.parseInt(st.nextToken());  
        //두 번째 토큰을 정수형으로 변환하여 변수 X에 저장  
  int[] arr = new int[N];  
        //길이가 N인 정수형 배열 선언  
  st = new StringTokenizer(bf.readLine()," ");  
        // 다음 한줄을 공백을 기준으로 나누어 저장  
  for(int i=0;i<N;i++) {  
            arr[i] = Integer.parseInt(st.nextToken());  
            //토큰을 정수형으로 변환하여 배열의 해당 인덱스에 저장  
  if(arr[i]<X){  
                System.out.print(arr[i]+" ");  
            }  
        }  
    }  
}
```

```java
import java.util.*;  
import java.io.*;  
public class Main {  
    public static void main(String[] args) throws IOException {  
        BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));  
  
        int N = Integer.parseInt(bf.readLine());  
        //bf.readLine()은 BufferedReader 객체인 bf를 사용하여 사용자로부터 한 줄을 읽어오는 메서드  
  
  int[] arr = new int[N]; //길이가 N인 배열 생성  
  
  StringTokenizer st = new StringTokenizer(bf.readLine()," ");  
  
        int min = Integer.parseInt(st.nextToken()); //첫번째 요소를 min으로 초기화  
  int max = min;  
        for(int i = 1; i<N; i++)  
        {  
            arr[i] = Integer.parseInt(st.nextToken());  
            if(arr[i]>max){  
                max = arr[i];  
            }  
            if(arr[i]<min)  
            {  
                min = arr[i];  
            }  
        }  
        System.out.println(min+" "+max);  
    }
```

```java
import java.util.*;  
import java.io.*;  
public class Main {  
    public static void main(String[] args) throws IOException {  
       BufferedReader bf = new BufferedReader(new InputStreamReader(System.in)); //선언  
  
  StringTokenizer st = new StringTokenizer(bf.readLine()," ");  
        //공백으로 토큰 나누기 위해  
  
  int N = Integer.parseInt(st.nextToken()); //String -> int 형변환 후 저장  
  int M = Integer.parseInt(st.nextToken());  
  
        int[] arr = new int[N]; //N 길이의 배열 저장  
  
  for(int i = 0; i < M; i++) //공을 넣는 횟수만큼 반복  
  {  
            st = new StringTokenizer(bf.readLine()," "); //공백으로 토큰 나누기 위해  
  int I = Integer.parseInt(st.nextToken()); //I에 첫번째 토큰 저장  
  int J = Integer.parseInt(st.nextToken()); //J에 두번째 토큰 저장  
  int K = Integer.parseInt(st.nextToken()); //K에 세번째 토큰 저장  
  
  for(int j=I-1; j<J; j++) //배열은 0부터 시작  
  {  
                arr[j] = K;  
            }  
        }  
        for(int k=0;k<N;k++)  
        {  
            System.out.print(arr[k]+" ");  
        }  
  
    }  
}
```

```java
           temp = arr[I-1];  
           arr[I-1] = arr[J-1];  
           arr[J-1] = temp;  
        }  
        for(int k=0;k<N;k++)  
        {  
            System.out.print(arr[k]+" ");  
        }
```

```java
import javax.swing.*;  
import java.util.*;  
import java.io.*;  
public class Main {  
    public static void main(String[] args) throws IOException {  
        BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));  
        StringTokenizer st = new StringTokenizer(bf.readLine());  
  
        int N = Integer.parseInt(st.nextToken());  
        int M = Integer.parseInt(st.nextToken());  
  
        int arr[] = new int[N+1];  
  
  
        for(int l=1;l<=N;l++)  
        {  
            arr[l]=l;  
        }  
  
        for(int i=1;i<=M;i++)  
        {  
            st = new StringTokenizer(bf.readLine()," ");  
  
            int I = Integer.parseInt(st.nextToken()); //연산 시작 인덱스  
  int J = Integer.parseInt(st.nextToken()); //연산 끝 인덱스  
  
  while(I<J) //시작 인덱스가 끝 인덱스보다 작을 때까지 반복  
  {  
                int temp = arr[I]; //시작 인덱스 값과 끝 인덱스 값을 교환  
  arr[I++] = arr[J]; //시작 인덱스 I를 증가 시키고  
  arr[J--] = temp; //끝 인덱스 J를 감소시켜 구간을 좁혀가서   
 //이러한 과정을 반복하여 구간 전체를 뒤집기  
  }  
        }  
        for(int k=1;k<=N;k++)  
        {  
            System.out.print(arr[k]+" ");  
        }  
    }  
}
```

```java
import javax.swing.*;  
import java.util.*;  
import java.io.*;  
public class Main {  
    public static void main(String[] args) throws IOException {  
        BufferedReader bf = new BufferedReader(new InputStreamReader(System.in));  
        StringTokenizer st = new StringTokenizer(bf.readLine());  
  
        int score = Integer.parseInt(st.nextToken());//과목수  
  double[] arr = new double[score]; //과목수 길이의 배열  
  
  double max =0, avg, sum =0;  
  
        st = new StringTokenizer(bf.readLine());  
        for(int i=0;i<score;i++) //과목수 길이만큼  
  {  
            arr[i] = Double.parseDouble(st.nextToken());  
  
            if (max < arr[i]) {  
                max = arr[i];  
            }  
        }  
        for(int j=0;j<score;j++)  
        {  
            arr[j] = (arr[j]/max)*100;  
            sum += arr[j];  
        }  
        avg = sum/score;  
        System.out.println(avg);  
  
    }  
}
```
